1.1 gitbook

Create the file "book.json"

cat > ./book.json << "EOF"
{
    "plugins" : [
        "toc",
    "expandable-chapters"
    ],
    "pluginsConfig": {
        "toc": {
            "addClass": true,
            "className": "toc"
        }
    }
}
EOF

install gitbook again:

gitbook install
